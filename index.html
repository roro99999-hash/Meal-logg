    <script>

        let meals = JSON.parse(localStorage.getItem('mealLog')) || [];

        updateTable();

        function logMeal() {

            const now = new Date();

            let diffString = "-";

            if (meals.length > 0) {

                const lastMealTime = new Date(meals[meals.length - 1].timestamp);

                const diffMs = now - lastMealTime;

               

                const diffMins = Math.floor(diffMs / 60000);

                const hours = Math.floor(diffMins / 60);

                const mins = diffMins % 60;

               

                diffString = `${hours}h ${mins}m`;

            } else {

                diffString = "First meal";

            }

            const mealData = {

                timestamp: now.toISOString(),

                displayTime: now.toLocaleString('en-GB', {

                    day: '2-digit', month: 'short',

                    hour: '2-digit', minute: '2-digit'

                }),

                timeSinceLast: diffString

            };

            meals.push(mealData);

            localStorage.setItem('mealLog', JSON.stringify(meals));

            updateTable();

        }

        function updateTable() {

            const tbody = document.getElementById('logTableBody');

            tbody.innerHTML = '';

            for (let i = meals.length - 1; i >= 0; i--) {

                const tr = document.createElement('tr');

                const tdTime = document.createElement('td');

                tdTime.textContent = meals[i].displayTime;

                const tdDiff = document.createElement('td');

                tdDiff.textContent = meals[i].timeSinceLast;

                tr.appendChild(tdTime);

                tr.appendChild(tdDiff);

                tbody.appendChild(tr);

            }

        }

        function exportToCSV() {

            if (meals.length === 0) {

                alert("No data to export.");

                return;

            }

            let csvContent = "Time;Time Since Last Meal\n";

            meals.forEach(meal => {

                csvContent += `"${meal.displayTime}";"${meal.timeSinceLast}"\n`;

            });

            const blob = new Blob(["\uFEFF" + csvContent], { type: 'text/csv;charset=utf-8;' });

            const link = document.createElement("a");

            const url = URL.createObjectURL(blob);

            link.setAttribute("href", url);

            link.setAttribute("download", "meal_log.csv");

            link.click();

        }

        function clearData() {

            if (confirm("Delete all history?")) {

                meals = [];

                localStorage.removeItem('mealLog');

                updateTable();

            }

        }
